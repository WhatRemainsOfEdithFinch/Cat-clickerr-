Buttons
	*catButton
		name:Click the cat
		desc:Click this adorable cat to get more cats!
		on click:anim icon wobble
		on click:yield 1 cat
		on click:if (have goldenFish and chance(1%)) yield 1 goldenFish
		icon:stuff/cat.png
		no text
		class:bigButton hasFlares
		icon class:shadowed
		tooltip origin:bottom
		tooltip class:red

Shinies
	*luckyCat
		on click:log Woop
		movement:onRight moveLeft fade bounce:0.05
		frequency:60
		frequency variation:30
		icon:stuff/luckyCat.png
		class:bigButton
		on click:
			$boost=1
			if (have clover) $boost=4
			if (chance(25%))
				$amount=max(7,random(cats:ps*60*1,cats:ps*60*3))*$boost
				toast The lucky cat grants you<//><b>[$amount] cats</b>.
				yield $amount cats
			else
				$amount=1*$boost
				toast The lucky cat grants you<//><b>[$amount] golden fish[s?$amount]</b>!
				yield $amount goldenFish
			end
		end
